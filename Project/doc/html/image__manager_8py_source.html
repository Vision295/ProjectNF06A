158 def get_new_pixel(self, _old_px:list, _mask:list) -> tuple:
159 """function used to change the value rgb of a pixel by applying a filter
160 @param _old_px is the rgb values of the pixel
161 @param _mask is the filter to apply to each pixel
162 @return is a tuple which corresponds to a modified pixel"""
163 return (
164 int(_old_px[0] * _mask[0][0] + _old_px[1] * _mask[0][1] + _old_px[2] * _mask[0][2]),
165 int(_old_px[0] * _mask[1][0] + _old_px[1] * _mask[1][1] + _old_px[2] * _mask[1][2]),
166 int(_old_px[0] * _mask[2][0] + _old_px[1] * _mask[2][1] + _old_px[2] * _mask[2][2])
167 )

196 def apply_custom_filter(self, _mask:tuple) -> None:
197 """function run when the button "filter" is pressed to apply a custom
198 filter
199 @param _mask is the filter to apply to each pixel"""
200 # we copy the image
201 self.filtered_image = self.image.copy()
202 # loop through the pixels of the image
203 for y in range(0, self.image_height - 1):
204 for x in range(0, self.image_width - 1):
205 # modify every pixel and save them in the copy of the original image
206 px = self.image.getpixel((x, y))
207 new_px = self.get_new_pixel(px, _mask)
208 self.filtered_image.putpixel((x, y), new_px)
209
210 # Save the modified image with a modified name
211 modified_name = self.current_path[:-4] + "filtered" + self.current_path[-4:] # Add "_filtered" before the file extension
212 self.filtered_image.save(modified_name)

295 def compress(self) -> None:
296 """function to compress an openned image and save the compress version in a folder"""
297 # Create the folder if it doesn't exist

304 filename = filename[::-1]
