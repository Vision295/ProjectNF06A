39 def resize_image_to_fit(self):
40 """function run in the open_image_png function to make it fit within the size of the window"""
41
42 # reduce the size of the image for it to fit within the size of the the window and fill it as much as possible
43 if self.image_width > self.image_height:
44 self.modified_image_size[0] = self.window_size
45 self.modified_image_size[1] = int(self.image_height * self.window_size / self.image_width)
46 else:
47 self.modified_image_size[1] = self.window_size
48 self.modified_image_size[0] = int(self.image_height * self.window_size / self.image_width)

139 def get_new_pixel(self, _old_px, _mask):
140 return (

163 def apply_custom_filter(self, _mask=None):
164 self.filtered_image = self.image.copy()
165 for y in range(0, self.image_height - 1):
166 for x in range(0, self.image_width - 1):
167 px = self.image.getpixel((x, y))
168 new_px = self.get_new_pixel(px, _mask)
169 self.filtered_image.putpixel((x, y), new_px)
