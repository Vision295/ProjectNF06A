23void Insertion_Sort(struct Node* Node_List, int n) {
24 int i=0;
25 while (i<n && Node_List[n-1].weight >= Node_List[i].weight) {
26 i++;

39void Visit_Tree_To_Get_Encoding(struct Node Node_to_visit, unsigned short conversion_array[256][2], unsigned short current_bin_code, unsigned char n_bits_curr_bit_code, int bit_to_add) {
40 int binary_code = 0;
41 //Depending of the bit_to add, we add it to the current bin code
42 if (bit_to_add == 0) {
43 binary_code = current_bin_code<<1;
44 //printf("\n%d %d", current_bin_code, current_bin_code<<1);
45 n_bits_curr_bit_code ++;

52 if (Node_to_visit.left != NULL) {

57 }
58 if (Node_to_visit.right == NULL && Node_to_visit.left == NULL) {
59 //We add the binary code at the correct index (corresponding to the value of the byte represented)
60 //conversion_array[i][0] is the size (in bits) of the code, and conversion_array[i][1] is the code in itself
